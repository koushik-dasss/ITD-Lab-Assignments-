CREATE DATABASE 2341004117_banking_db;
USE 2341004117_banking_db;

CREATE TABLE BRANCH (
    BRANCH_CODE VARCHAR(5),
    BRANCH_NAME VARCHAR(30),
    CITY VARCHAR(20)
);

CREATE TABLE CUSTOMER (
    CUST_NO VARCHAR(5),
    CUST_NAME VARCHAR(30),
    ADDRESS VARCHAR(40),
    PHONE NUMERIC(10)
);

CREATE TABLE ACCOUNT (
    ACC_NO VARCHAR(10),
    CUST_NO VARCHAR(5),
    BRANCH_CODE VARCHAR(5),
    ACC_TYPE VARCHAR(10),
    BALANCE NUMERIC(10)
);

CREATE TABLE LOAN (
    LOAN_NO VARCHAR(10),
    CUST_NO VARCHAR(5),
    BRANCH_CODE VARCHAR(5),
    AMOUNT NUMERIC(10)
);

CREATE TABLE TRANSACTION_DETAILS (
    TRANS_ID VARCHAR(10),
    ACC_NO VARCHAR(10),
    TRANS_TYPE VARCHAR(10),
    TRANS_AMOUNT NUMERIC(10),
    TRANS_DATE DATE
);
